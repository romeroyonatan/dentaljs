        $scope.save = (patient) ->
      if patient.civil_status?
        patient.civil_status = parseInt patient.civil_status
      person = new Person patient
      if not $routeParams.slug
        person.$save ->
          $location.path "/patients/#{person.slug}"
      else
        person.slug = $routeParams.slug
        person.$update ->
          $location.path "/patients/#{person.slug}"
