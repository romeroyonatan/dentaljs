    balance

    Calculate the person’s balance

      balance: (req, res, next) ->
    Accounting.find person: req.params.person, (err, list)->
      if err
        console.error err
        return next err
      balance = 0
      for account in list
        balance += account.assets
        balance -= account.debit
      res.send balance: balance
